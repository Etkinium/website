<script>
  // PUAN TOGGLE MANTIĞI
  const puanToggle = document.getElementById("puanToggle");
  const hotelItems = document.querySelectorAll(".hotel-item");

  function formatTl(value) {
    return (
      new Intl.NumberFormat("tr-TR").format(Number(value)) + " TL"
    );
  }

  function formatPuan(value) {
    return (
      new Intl.NumberFormat("tr-TR").format(Number(value)) +
      " Etkinium Puan"
    );
  }

  function updatePrices() {
    const usePoints = puanToggle.checked;
    hotelItems.forEach((item) => {
      const price = item.getAttribute("data-price");
      const points = item.getAttribute("data-points");
      const tlEl = item.querySelector(".price-tl");
      const puanEl = item.querySelector(".price-puan");

      if (!tlEl || !puanEl) return;

      if (usePoints) {
        tlEl.style.display = "none";
        puanEl.style.display = "block";
        puanEl.textContent =
          formatPuan(points) +
          " ile bu konaklama tamamen puanla ödenebilir.";
      } else {
        tlEl.style.display = "block";
        puanEl.style.display = "block";
        tlEl.textContent = formatTl(price);
        puanEl.textContent =
          "Yaklaşık " + formatPuan(points) + " değerinde puan karşılığı.";
      }
    });
  }

  if (puanToggle) {
    puanToggle.addEventListener("change", updatePrices);
    updatePrices();
  }

  // SLIDER OTOMATİK GEÇİŞ
  const slides = document.querySelectorAll(".konaklama-slide");
  const dots = document.querySelectorAll(".konaklama-dot");
  let currentSlide = 0;

  function showSlide(index) {
    slides.forEach((s, i) => {
      s.classList.toggle("active", i === index);
    });
    dots.forEach((d, i) => {
      d.classList.toggle("active", i === index);
    });
    currentSlide = index;
  }

  dots.forEach((dot, index) => {
    dot.addEventListener("click", () => showSlide(index));
  });

  setInterval(() => {
    const next = (currentSlide + 1) % slides.length;
    showSlide(next);
  }, 4000);
</script>
